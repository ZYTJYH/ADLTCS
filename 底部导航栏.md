class BottomBarFragment extends Fragment
{
   private ImageView startMainActivityImage;
   private ImageView startOrderActivityImage;
   private ImageView startMyActivityImage;
   onCreateView(){
      startMainActivityImage=(ImageView)findViewById(R.id.startMainActivityImage);
      startOrderActivityImage=(ImageView)findViewById(R.id.startOrderActivityImage);
      startMyActivityImage=(ImageView)findViewById(R.id.startMyActivityImage);
      startMainActivityImage.setOnClickListener(OnClickListener(){
          onClick(){
            if(!getRunningActivityName.equals("MainActivity.class")){
            BottomBarFragment.this.startActivity(new Intent(BottomBarFragment.this,MainActivity.class));
            startMainActivityImage.setImageDrawable(getResources().getDrawable(--当前页面--));
            startOrderActivityImage.setImageDrawable(getResources().getDrawable(--非当前页面--));
            startMyActivityImage.setImageDrawable(getResources().getDrawable(--非当前页面--));
            }
          }
      }
      
      startOrderActivityImage.setOnClickListener(OnClickListener(){
          onClick(){
           BottomBarFragment.this.startActivity(
            if(!getRunningActivityName.equals("OrderActivity.class")){
            BottomBarFragment.this.startActivity(new Intent(BottomBarFragment.this,OrderActivity.class));
            startMainActivityImage.setImageDrawable(getResources().getDrawable(--非当前页面--));
            startOrderActivityImage.setImageDrawable(getResources().getDrawable(--当前页面--));
            startMyActivityImage.setImageDrawable(getResources().getDrawable(--非当前页面--));
            }
          }
      }
      
      startGoActivityImage.setOnClickListener(OnClickListener(){
          onClick(){
           BottomBarFragment.this.startActivity(
            if(!getRunningActivityName.equals("OrderActivity.class")){
            BottomBarFragment.this.startActivity(new Intent(BottomBarFragment.this,OrderActivity.class));
            startMainActivityImage.setImageDrawable(getResources().getDrawable(--非当前页面--));
            startOrderActivityImage.setImageDrawable(getResources().getDrawable(--非当前页面--));
            startMyActivityImage.setImageDrawable(getResources().getDrawable(--当前页面--));
            }
          }
      }
   }
   private String getRunningActivityName() {  
        String contextString = getActivity().toString();  
        return contextString.substring(contextString.lastIndexOf(".") + 1, contextString.indexOf("@"));  
   }
     
}
